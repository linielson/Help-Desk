def total_tempo_trabalhado
  total = 0
  tarefa_horas.each do |hora|
    total += (hora.fim - hora.inicio)
  end
  total
end

def validar_cliente_e_projeto

    errors.add(:projeto_id, 'deve ser selecionado um projeto ou um cliente!')

  end
end
